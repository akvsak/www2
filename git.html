~/.git/config file add the following aliases to switch users 

[user]
name = 3AA
email = 3@a.com

[alias]
set1 = !git config user.name '1AA' && git config user.email '1@a.com'
set2 = "config user.name '2Arnaud Rinquin' user.email '2arnaud.rinquin@wopata.com'"
set3 = !git config user.name '3AA' && git config user.email '3@a.com'
